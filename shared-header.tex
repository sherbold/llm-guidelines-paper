% Cross-reference helpers with optional ID prefix
% Usage: \makepaperref{label}{Name}{ID} → "Name", \makepaperref[id]{label}{Name}{ID} → "ID: Name"
\newcommand{\makepaperref}[4][]{%
  \if\relax\detokenize{#1}\relax
    \hyperref[#2]{\emph{#3}}%
  \else
    \hyperref[#2]{\emph{#4: #3}}%
  \fi\xspace}
\newcommand{\makewebref}[4][]{%
  \if\relax\detokenize{#1}\relax
    \href{#2}{\emph{#3}}%
  \else
    \href{#2}{\emph{#4: #3}}%
  \fi\xspace}
